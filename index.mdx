<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-3xl font-bold text-gray-900 dark:text-white mb-3">
    Quick Start
  </h2>
  <p className="text-lg text-gray-600 dark:text-gray-400 mb-8">
    Get AdMob Plus running in your app in minutes
  </p>
  
  <Steps>
    <Step title="Install the package">
      Choose the package for your framework:
      
      <CodeGroup>
      ```bash Cordova
      cordova plugin add admob-plus-cordova
      ```
      ```bash Capacitor
      npm install @admob-plus/capacitor
      ```
      ```bash React Native
      npm install @admob-plus/react-native
      ```
      ```bash Ionic
      npm install @admob-plus/ionic
      ```
      </CodeGroup>
    </Step>
    
    <Step title="Initialize AdMob">
      Start AdMob in your app's initialization code:
      
      ```typescript
      import { AdMob } from 'admob-plus-cordova';
      
      await AdMob.start();
      ```
    </Step>
    
    <Step title="Create and show an ad">
      Create a banner ad and display it on screen:
      
      ```typescript
      const banner = new AdMob.BannerAd({
        adUnitId: 'ca-app-pub-xxx/xxx',
      });
      
      await banner.show();
      ```
      
      <Note>
        Use test ad unit IDs during development. See the [Test Ads guide](/guides/test-ads) for test IDs.
      </Note>
    </Step>
  </Steps>
</div>
